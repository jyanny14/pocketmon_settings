# Policy
#
# 1. AI training crawlers — block everything. Our dataset derives from
#    serebii / PokeAPI; we don't want Nintendo IP baked into model weights.
# 2. User-triggered AI browsing — allow. When a user pastes a URL into
#    ChatGPT / Claude.ai / Perplexity, the fetch should succeed.
# 3. Unknown crawlers — deny HTML pages (also noindex via <meta>), but
#    allow the data JSON + llms.txt so AI tools with non-standard UAs
#    can still get the data when a user pastes the link.

# ── AI training crawlers — disallow ─────────────────────────────
User-agent: GPTBot
Disallow: /

User-agent: anthropic-ai
Disallow: /

User-agent: Google-Extended
Disallow: /

User-agent: CCBot
Disallow: /

User-agent: Bytespider
Disallow: /

User-agent: Meta-ExternalAgent
Disallow: /

User-agent: Applebot-Extended
Disallow: /

# ── User-triggered AI browsing — allow ──────────────────────────
User-agent: ChatGPT-User
Allow: /

User-agent: OAI-SearchBot
Allow: /

User-agent: ClaudeBot
Allow: /

User-agent: Claude-Web
Allow: /

User-agent: PerplexityBot
Allow: /

User-agent: Perplexity-User
Allow: /

# ── Default: block HTML indexing, allow data files ──────────────
User-agent: *
Disallow: /
Allow: /data/
Allow: /llms.txt
Allow: /robots.txt
